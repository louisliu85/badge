import React, { useState, useRef, useEffect } from 'react';

import { User, Plus, Image as ImageIcon, X, Check, Search, Upload, RefreshCw, Save, Trash2, Loader2, Settings, Lock, Unlock } from 'lucide-react';

import { initializeApp } from 'firebase/app';

import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth';

import { getFirestore, collection, doc, setDoc, onSnapshot, query, deleteDoc } from 'firebase/firestore';


// ==========================================

// 1. Firebase 配置與初始化

// ==========================================

const firebaseConfig = JSON.parse(__firebase_config);

const app = initializeApp(firebaseConfig);

const auth = getAuth(app);

const db = getFirestore(app);


// 關鍵修正：使用系統配發的 ID 以確保符合安全規則。

// 只要您在不同裝置開啟同一個專案，appId 就會保持一致，實現雲端同步。

const appId = typeof __app_id !== 'undefined' ? __app_id : 'decathlon-tag-generator-default'; 


// 管理員密碼設定

const ADMIN_PASSWORD = "1204@1769";


// 預設配置

const FALLBACK_LOGO = "https://upload.wikimedia.org/wikipedia/commons/b/bc/Decathlon_-_logo_%28France%2C_2024%29.svg";

const DEFAULT_ICONS = [

  { id: 'default_running', name: '無', url: '' },

];

const FIXED_SLOGAN = "你的滿意，我們在意";


const App = () => {

  // --- 狀態管理 ---

  const [user, setUser] = useState(null);

  const [name, setName] = useState('Louis');

  const [logoUrl, setLogoUrl] = useState(FALLBACK_LOGO);

  const [sportIconL, setSportIconL] = useState('');

  const [sportIconR, setSportIconR] = useState('');

  const [isGenerating, setIsGenerating] = useState(false);

  const [libsLoaded, setLibsLoaded] = useState(false);

  const [showSelector, setShowSelector] = useState(null); 

  const [searchTerm, setSearchTerm] = useState('');

  const [cloudIcons, setCloudIcons] = useState([]);

  const [isSaving, setIsSaving] = useState(false);

  const [isLogoSaving, setIsLogoSaving] = useState(false);

  

  // 管理選單與權限狀態

  const [showSettings, setShowSettings] = useState(false);

  const [isUnlocked, setIsUnlocked] = useState(false);

  const [passwordInput, setPasswordInput] = useState('');

  const [passwordError, setPasswordError] = useState(false);


  const cardRef = useRef(null);

  const iconFileInputRef = useRef(null);

  const logoFileInputRef = useRef(null);


  // --- 認證與初始載入 (符合 Rule 3: 必須先處理認證) ---

  useEffect(() => {

    const loadScript = (url) => {

      return new Promise((resolve) => {

        const script = document.createElement('script');

        script.src = url;

        script.onload = resolve;

        document.head.appendChild(script);

      });

    };


    Promise.all([

      loadScript('https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js'),

      loadScript('https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js')

    ]).then(() => setLibsLoaded(true));


    const initAuth = async () => {

      try {

        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

          await signInWithCustomToken(auth, __initial_auth_token);

        } else {

          await signInAnonymously(auth);

        }

      } catch (err) {

        console.error("Auth Error:", err);

      }

    };

    initAuth();

    const unsubscribe = onAuthStateChanged(auth, setUser);

    return () => unsubscribe();

  }, []);


  // --- Firestore 資料同步 (符合 Rule 1 & 2) ---

  useEffect(() => {

    if (!user) return;


    // A. 監聽雲端圖示

    const iconsCol = collection(db, 'artifacts', appId, 'public', 'data', 'sport_icons');

    const unsubscribeIcons = onSnapshot(iconsCol, (snapshot) => {

      const icons = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

      setCloudIcons(icons);

    }, (error) => console.error("Icons Fetch Error:", error));


    // B. 監聽 Logo

    const logoDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'branding', 'logo');

    const unsubscribeLogo = onSnapshot(logoDocRef, (docSnap) => {

      if (docSnap.exists() && docSnap.data().url) {

        setLogoUrl(docSnap.data().url);

      } else {

        setLogoUrl(FALLBACK_LOGO);

      }

    }, (error) => console.error("Logo Fetch Error:", error));


    return () => {

      unsubscribeIcons();

      unsubscribeLogo();

    };

  }, [user]);


  // --- 核心邏輯 ---

  const getNameFontSize = (text) => {

    const maxFontSize = 40;

    const minFontSize = 25;

    const threshold = 5;

    if (!text || text.length <= threshold) return maxFontSize;

    const scaledSize = (threshold / text.length) * maxFontSize;

    return Math.max(minFontSize, scaledSize);

  };


  const dynamicFontSize = getNameFontSize(name);

  const hasDescenderLetters = /[gjypq]/i.test(name);


  // 密碼驗證

  const handleUnlock = () => {

    if (passwordInput === ADMIN_PASSWORD) {

      setIsUnlocked(true);

      setPasswordError(false);

      setPasswordInput('');

    } else {

      setPasswordError(true);

      setTimeout(() => setPasswordError(false), 2000);

    }

  };


  // Logo 上傳

  const handleLogoUpload = async (e) => {

    const file = e.target.files[0];

    if (!file || !user) return;

    setIsLogoSaving(true);

    const reader = new FileReader();

    reader.onloadend = async () => {

      try {

        const logoDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'branding', 'logo');

        await setDoc(logoDocRef, { url: reader.result, updatedAt: Date.now() });

      } catch (err) {

        console.error("Logo 上傳失敗:", err);

      } finally {

        setIsLogoSaving(false);

      }

    };

    reader.readAsDataURL(file);

  };


  // 運動圖示上傳

  const handleCloudIconUpload = async (e) => {

    const file = e.target.files[0];

    if (!file || !user) return;

    setIsSaving(true);

    const fileName = file.name.split('.').slice(0, -1).join('.') || "未命名圖示";

    const reader = new FileReader();

    reader.onloadend = async () => {

      try {

        const newIconId = crypto.randomUUID();

        const iconDoc = doc(db, 'artifacts', appId, 'public', 'data', 'sport_icons', newIconId);

        await setDoc(iconDoc, { name: fileName, url: reader.result, createdAt: Date.now() });

      } catch (err) {

        console.error("圖示上傳失敗:", err);

      } finally {

        setIsSaving(false);

        if (iconFileInputRef.current) iconFileInputRef.current.value = "";

      }

    };

    reader.readAsDataURL(file);

  };


  const deleteIcon = async (e, id) => {

    e.stopPropagation();

    if (!user || !isUnlocked) return;

    try {

      await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'sport_icons', id));

    } catch (err) {

      console.error("刪除失敗:", err);

    }

  };


  const downloadPDF = async () => {

    if (!cardRef.current || !libsLoaded) return;

    setIsGenerating(true);

    await new Promise(resolve => setTimeout(resolve, 500));

    try {

      const canvas = await window.html2canvas(cardRef.current, { scale: 4, useCORS: true, backgroundColor: '#ffffff' });

      const imgData = canvas.toDataURL('image/png', 1.0);

      const pdf = new window.jspdf.jsPDF('p', 'mm', 'a4');

      const cardWidth = 55;

      const cardHeight = 85;

      const gap = 15;

      const totalWidth = (cardWidth * 2) + gap;

      const startX = (210 - totalWidth) / 2;

      const startY = 40;


      const drawCropMarks = (x, y) => {

        pdf.setDrawColor(200);

        pdf.setLineWidth(0.1);

        const len = 5; const offset = 2;

        pdf.line(x - len - offset, y, x - offset, y); pdf.line(x, y - len - offset, x, y - offset);

        pdf.line(x + cardWidth + offset, y, x + cardWidth + len + offset, y); pdf.line(x + cardWidth, y - len - offset, x + cardWidth, y - offset);

        pdf.line(x - len - offset, y + cardHeight, x - offset, y + cardHeight); pdf.line(x, y + cardHeight + offset, x, y + cardHeight + len + offset);

        pdf.line(x + cardWidth + offset, y + cardHeight, x + cardWidth + len + offset, y + cardHeight); pdf.line(x + cardWidth, y + cardHeight + offset, x + cardWidth, y + cardHeight + len + offset);

      };


      const x1 = startX;

      drawCropMarks(x1, startY);

      pdf.addImage(imgData, 'PNG', x1, startY, cardWidth, cardHeight, undefined, 'FAST');

      const x2 = startX + cardWidth + gap;

      drawCropMarks(x2, startY);

      pdf.addImage(imgData, 'PNG', x2, startY, cardWidth, cardHeight, undefined, 'FAST');

      pdf.save(`名牌_${name}.pdf`);

    } catch (e) {

      console.error("生成 PDF 失敗:", e);

    } finally {

      setIsGenerating(false);

    }

  };


  const allIcons = [...DEFAULT_ICONS, ...cloudIcons];

  const filteredIcons = allIcons.filter(icon => icon.name.toLowerCase().includes(searchTerm.toLowerCase()));


  return (

    <div className="min-h-screen bg-slate-100 p-4 md:p-8 text-slate-800" style={{ fontFamily: 'sans-serif' }}>

      <div className="max-w-6xl mx-auto grid grid-cols-1 lg:grid-cols-12 gap-8">

        

        {/* 控制面板 */}

        <div className="lg:col-span-4 space-y-6 bg-white p-6 rounded-2xl shadow-sm border border-slate-200 h-fit">

          <div className="border-b pb-4 flex justify-between items-start">

            <div>

              <h1 className="text-xl font-bold text-[#2E4496] flex items-center gap-2">名牌產生器</h1>

              <p className="text-xs text-slate-400 mt-1 font-medium">

                {isUnlocked ? '管理員模式已啟動' : '輕鬆幫你做出名牌！'}

              </p>

            </div>

            <button 

              onClick={() => { setShowSettings(!showSettings); if (showSettings) setIsUnlocked(false); }}

              className={`p-2 rounded-lg transition-all ${showSettings ? 'bg-[#2E4496] text-white rotate-90' : 'bg-slate-50 text-slate-400 hover:bg-slate-100'}`}

            >

              <Settings size={22} />

            </button>

          </div>

          

          <div className="space-y-4">

            {/* 齒輪管理選單 */}

            {showSettings && (

              <div className="p-4 bg-slate-50 rounded-2xl border border-slate-200 space-y-4 animate-in fade-in slide-in-from-top-4 duration-300">

                <div className="flex items-center justify-between">

                  <h4 className="text-sm font-bold text-slate-700 flex items-center gap-2">

                    {isUnlocked ? <Unlock size={16} className="text-green-500" /> : <Lock size={16} className="text-slate-400" />}

                    遇到問題？請聯絡小路

                  </h4>

                  {isUnlocked && <button onClick={() => setIsUnlocked(false)} className="text-[10px] text-[#2E4496] underline">重新上鎖</button>}

                </div>

                {!isUnlocked ? (

                  <div className="space-y-2">

                    <input 

                      type="password" placeholder="輸入管理密碼..." 

                      className={`w-full px-3 py-2 text-sm border rounded-lg outline-none transition-all ${passwordError ? 'border-red-500 bg-red-50' : 'border-slate-200'}`}

                      value={passwordInput} onChange={(e) => setPasswordInput(e.target.value)}

                      onKeyDown={(e) => e.key === 'Enter' && handleUnlock()}

                    />

                    <button onClick={handleUnlock} className="w-full py-2 bg-[#2E4496] text-white rounded-lg text-xs font-bold hover:bg-[#1d2d66]">驗證解鎖</button>

                  </div>

                ) : (

                  <div className="space-y-4 animate-in zoom-in-95 duration-200">

                    <div className="pt-2 border-t border-slate-200">

                      <p className="text-[11px] font-bold text-slate-500 mb-2 uppercase tracking-wider">1. 品牌 Logo 管理</p>

                      <button onClick={() => logoFileInputRef.current?.click()} disabled={isLogoSaving} className="w-full py-2.5 bg-white border border-[#2E4496] text-[#2E4496] rounded-lg text-xs font-bold flex items-center justify-center gap-2">

                        {isLogoSaving ? <Loader2 size={14} className="animate-spin" /> : <ImageIcon size={14} />} 重新上傳品牌 Logo

                      </button>

                    </div>

                    <div className="pt-2 border-t border-slate-200">

                      <p className="text-[11px] font-bold text-slate-500 mb-2 uppercase tracking-wider">2. 運動圖示庫管理</p>

                      <button onClick={() => iconFileInputRef.current?.click()} disabled={isSaving} className="w-full py-2.5 bg-green-600 text-white rounded-lg text-xs font-bold flex items-center justify-center gap-2 shadow-sm">

                        {isSaving ? <Loader2 size={14} className="animate-spin" /> : <Upload size={14} />} 上傳新圖示至選單

                      </button>

                    </div>

                  </div>

                )}

              </div>

            )}


            <div>

              <label className="block text-sm font-medium mb-1 text-slate-600">員工姓名</label>

              <div className="relative">

                <User className="absolute left-3 top-2.5 h-4 w-4 text-slate-400" />

                <input

                  type="text"

                  value={name}

                  onChange={(e) => setName(e.target.value)}

                  className="w-full pl-10 pr-4 py-2 bg-slate-50 border border-slate-200 rounded-lg focus:ring-1 focus:ring-[#149B65] outline-none font-bold text-[#149B65]"

                  placeholder="請輸入姓名"

                />

              </div>

              <p className="text-[10px] text-slate-400 mt-1">※ 遇「空格」將自動換行</p>

            </div>


            <div className="grid grid-cols-2 gap-4">

              <div className="space-y-1">

                <label className="block text-sm font-medium text-slate-600">運動 1</label>

                <button onClick={() => setShowSelector('L')} className="w-full h-24 bg-slate-50 border-2 border-dashed border-slate-200 rounded-xl flex items-center justify-center overflow-hidden p-2 hover:border-[#149B65] transition-colors shadow-inner">

                  {sportIconL ? <img src={sportIconL} className="h-full object-contain" /> : <Plus className="text-slate-300" />}

                </button>

              </div>

              <div className="space-y-1">

                <label className="block text-sm font-medium text-slate-600">運動 2</label>

                <button onClick={() => setShowSelector('R')} className="w-full h-24 bg-slate-50 border-2 border-dashed border-slate-200 rounded-xl flex items-center justify-center overflow-hidden p-2 hover:border-[#149B65] transition-colors shadow-inner">

                  {sportIconR ? <img src={sportIconR} className="h-full object-contain" /> : <Plus className="text-slate-300" />}

                </button>

              </div>

            </div>

          </div>


          <button onClick={downloadPDF} disabled={isGenerating || !libsLoaded} className="w-full mt-6 bg-[#2E4496] text-white py-4 rounded-xl font-bold shadow-lg flex items-center justify-center gap-2 disabled:opacity-50 hover:bg-[#1d2d66] transition-all active:scale-95">

            {isGenerating ? <Loader2 className="animate-spin" /> : "下載 PDF"}

          </button>

        </div>


        {/* 右側預覽區 */}

        <div className="lg:col-span-8 flex flex-col items-center justify-center bg-slate-200 rounded-3xl p-10 min-h-[600px] overflow-auto">

          <div ref={cardRef} className="bg-white shadow-2xl flex flex-col items-start" style={{ width: '55mm', height: '85mm', border: '0.1mm solid #e2e8f0', overflow: 'hidden' }}>

            <div className="pt-[8mm] px-[6mm] w-full h-[16mm]">

              <img src={logoUrl} className="max-w-[50%] h-auto object-contain" crossOrigin="anonymous" alt="Brand Logo" />

            </div>

            <div className="flex-1 px-[6mm] w-full pt-[3mm] flex flex-col text-left">

              <p style={{ color: '#2E4496', fontSize: '20pt', fontWeight: 700, margin: 0, lineHeight: 1.1 }}>你好，我是</p>

              

              <h2 

                style={{ 

                  color: '#149B65', 

                  fontSize: `${dynamicFontSize}pt`, 

                  fontWeight: 700,

                  lineHeight: '1.05', 

                  marginTop: name.length < 6 ? '-1mm' : '1mm',

                  marginBottom: (name.length < 6 && hasDescenderLetters) ? '6mm' : '4mm',

                  textAlign: 'left',

                  wordBreak: 'break-word',

                  display: 'block'

                }} 

              >

                {name ? name.split(' ').map((line, i) => (

                  <span key={i} style={{ display: 'block' }}>{line}</span>

                )) : "Louis"}

              </h2>

              

              <p style={{ color: '#2E4496', fontSize: '13pt', fontWeight: 500, margin: 0, marginTop: 'auto', paddingBottom: '5mm' }}>{FIXED_SLOGAN}</p>

            </div>

            <div className="h-[28mm] px-[6mm] pb-[8mm] w-full flex gap-6 items-center">

              <div className="w-[16mm] h-[16mm] flex items-center justify-center">

                {sportIconL && <img src={sportIconL} className="max-w-full max-h-full object-contain" crossOrigin="anonymous" />}

              </div>

              <div className="w-[16mm] h-[16mm] flex items-center justify-center">

                {sportIconR && <img src={sportIconR} className="max-w-full max-h-full object-contain" crossOrigin="anonymous" />}

              </div>

            </div>

          </div>

          <p className="mt-6 text-slate-500 text-sm font-medium text-center">

            即時預覽僅供參考，實際依 PDF 為主。

          </p>

        </div>

      </div>


      {/* 圖示選單彈窗 */}

      {showSelector && (

        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/60 backdrop-blur-sm">

          <div className="bg-white w-full max-w-2xl rounded-3xl shadow-2xl overflow-hidden flex flex-col max-h-[85vh]">

            <div className="p-6 border-b flex justify-between items-center bg-slate-50">

              <div>

                <h3 className="font-bold text-lg text-[#2E4496]">運動圖示選單</h3>

                <p className="text-xs text-slate-400">目前共載入 {allIcons.length} 個項目</p>

              </div>

              <button onClick={() => setShowSelector(null)} className="p-2 hover:bg-slate-200 rounded-full transition-colors"><X size={20} /></button>

            </div>

            <div className="p-4 border-b">

              <div className="relative">

                <Search className="absolute left-4 top-3 h-4 w-4 text-slate-400" />

                <input 

                  type="text" placeholder="搜尋運動項目..." 

                  className="w-full pl-11 pr-4 py-2 bg-slate-100 rounded-xl outline-none"

                  value={searchTerm} onChange={(e) => setSearchTerm(e.target.value)}

                />

              </div>

            </div>

            <div className="flex-1 overflow-y-auto p-6 grid grid-cols-3 sm:grid-cols-4 md:grid-cols-5 gap-4">

              {filteredIcons.map((icon) => (

                <div key={icon.id} className="relative group">

                  <button

                    onClick={() => { if (showSelector === 'L') setSportIconL(icon.url); if (showSelector === 'R') setSportIconR(icon.url); setShowSelector(null); }}

                    className="w-full flex flex-col items-center gap-2 p-1 rounded-2xl hover:bg-green-50 border border-transparent hover:border-green-200 transition-all"

                  >

                    <div className="w-full aspect-square bg-slate-50 rounded-xl flex items-center justify-center p-3 group-hover:scale-105 transition-transform shadow-sm border border-slate-100 overflow-hidden">

                      {icon.url ? (

                        <img src={icon.url} alt={icon.name} className="max-w-full max-h-full object-contain" />

                      ) : (

                        <span className="text-[10px] text-slate-300">N/A</span>

                      )}

                    </div>

                    <span className="text-[10px] font-bold text-slate-500 text-center leading-tight truncate w-full mt-1 px-1">{icon.name}</span>

                  </button>

                  {isUnlocked && !icon.id.toString().startsWith('default') && (

                    <button onClick={(e) => deleteIcon(e, icon.id)} className="absolute -top-1 -right-1 p-1 bg-red-100 text-red-500 rounded-full opacity-0 group-hover:opacity-100 transition-opacity hover:bg-red-200 shadow-sm">

                      <Trash2 size={12} />

                    </button>

                  )}

                </div>

              ))}

            </div>

          </div>

        </div>

      )}


      <input type="file" id="logo-upload" ref={logoFileInputRef} className="hidden" accept="image/*" onChange={handleLogoUpload} />

      <input type="file" id="icon-upload" ref={iconFileInputRef} className="hidden" accept="image/*" onChange={handleCloudIconUpload} />

    </div>

  );

};
